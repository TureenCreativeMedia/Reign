identifiers:

	Note that unsigned variable types persist identifiers:
		eg. an unsigned int = i_VariableName

	Enums should not apply to this rule.

	s_ = string
	f_ = float
	i_ = integer
	c_ = char
	b_ = boolean
	a_ = action
	f_ = file
	u_ = anything unity related
	v_ = any vector
	r_ = anything reign related
	e_ = external
	d_ = double
	dn_ = dictionary
